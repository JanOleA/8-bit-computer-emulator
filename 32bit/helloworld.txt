char = 255
__textloc = 254
text = 240

.__textloc  = 240
.text     = " Hello World!"

start:
  DIS 1                 ; 0000 0001 - clear display
  JSR instruction
  DIS 6                 ; 0000 0110 - Set entry mode, cursor direction to increment
  JSR instruction       
  DIS 15                ; set on, display cursor, cursor blink
  JSR instruction

helloworld:
  JSR get_next_char
  CPI 0
  JPZ end_1
  JSR write_char
  JMP helloworld
end_1:
  DIS 7                 ; 0000 0110 - Set entry mode, cursor direction to increment and display shift
  JSR instruction
  LDI 240
  STA .__textloc
helloworld_2:
  JSR get_next_char
  CPI 0
  JPZ end_1
  JSR write_char
  JMP helloworld_2
end_printloop:
  HLT  

get_next_char:  ; gets the next character from .__textloc and increases .__textloc by 1
  LDA .__textloc
  PHA
  ADI 1
  STA .__textloc
  LAS         ; get the value from .__textloc into A
  STA .char
  RET

write_char:
  LDD .char
  DIC 0
  DIC 64      ; set read
  DIC 192     ; set read + enable
  DIC 0
  RET

instruction:
  DIC 0
  DIC 128     ; set enable
  DIC 0
  RET
